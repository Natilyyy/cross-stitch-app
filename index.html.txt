            <input type="range" id="pixelSize" min="4" max="30" value="10">

        <div class="control-group">

<script>
    const upload = document.getElementById('upload');
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    const pixelSlider = document.getElementById('pixelSize');
    const speedSlider = document.getElementById('speed');
    
    let img = new Image();
    let animationId;
    let offset = 0;

    upload.onchange = function(e) {
        const reader = new FileReader();
        reader.onload = function(event) {
            img.onload = function() {
                initCanvas();
                animate();
            }
            img.src = event.target.result;
        }
        reader.readAsDataURL(e.target.files[0]);
    };

    function initCanvas() {
        // 限制最大显示宽度
        const maxW = 700;
        const scale = maxW / img.width;
        canvas.width = maxW;
        canvas.height = img.height * scale;
    }

    function drawStitch() {
        const size = parseInt(pixelSlider.value);
        document.getElementById('pixelVal').innerText = size;
        
        const w = canvas.width;
        const h = canvas.height;

        // 1. 绘制原始图片缩小版（实现像素化基础）
        ctx.drawImage(img, 0, 0, w, h);
        const imgData = ctx.getImageData(0, 0, w, h).data;
        ctx.clearRect(0, 0, w, h);

        // 2. 遍历像素并绘制“十字绣”单元
        for (let y = 0; y < h; y += size) {
            for (let x = 0; x < w; x += size) {
                const i = (y * w + x) * 4;
                const r = imgData[i];
                const g = imgData[i+1];
                const b = imgData[i+2];

                // 增加动态色彩偏移（动画效果核心）
                const pulse = Math.sin(Date.now() * 0.002 * speedSlider.value + (x+y)*0.01) * 10;
                
                ctx.fillStyle = `rgb(${r+pulse},${g+pulse},${b+pulse})`;
                
                // 绘制十字绣小方块
                ctx.fillRect(x + 1, y + 1, size - 2, size - 2);
                
                // 绘制淡淡的交叉线模拟缝线
                ctx.strokeStyle = `rgba(255,255,255,0.2)`;
                ctx.beginPath();
                ctx.moveTo(x, y); ctx.lineTo(x + size, y + size);
                ctx.moveTo(x + size, y); ctx.lineTo(x, y + size);
                ctx.stroke();
            }
        }
    }

    function animate() {
        drawStitch();
        animationId = requestAnimationFrame(animate);
    }

    function downloadImage() {
        const link = document.createElement('a');
        link.download = 'cross-stitch.png';
        link.href = canvas.toDataURL();
        link.click();
    }
</script>
